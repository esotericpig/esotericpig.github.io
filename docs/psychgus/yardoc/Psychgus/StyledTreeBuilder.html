def alias(*)
  node = super

  @stylers.each do |styler|
    styler.style(sniffer,node)
    styler.style_alias(sniffer,node)
  end

  @sniffer.add_alias(node)

  return node
end

def pop_styler(count = 1)
  return nil if count == 0
  return @stylers.pop if count == 1

  return @stylers.pop(count)
end
